\noindent \textbf{Questions}

Evaluate the following LISP code.

\begin{enumerate}

\item
\begin{verbatim}
(DIV 24 (SUB 2 5))
\end{verbatim}

\item
\begin{verbatim}
(ADD (MULT 48 40)
     (SUB (SQUARE 9) 1)
     (MULT 3 5))
\end{verbatim}

\item
\begin{verbatim}
(SETQ a 'hello)
(ATOM a)
\end{verbatim}

\item
\begin{verbatim}
(SETQ b '((what) is a list "?"))
(CONS (CAR b) (CDR b))
(SETQ (CAR b) (CAR (CAR b)))
\end{verbatim}

\item
\begin{verbatim}
(SET 'c '(21 22 23 50))
(SETQ (CDR (CDR c)) (CONS 33 NIL))
(REVERSE c)
\end{verbatim}

\end{enumerate}
